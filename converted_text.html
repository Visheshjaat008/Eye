<script>
// Elements
const permissionSection = document.getElementById('permissionSection');
const uploadSection = document.getElementById('uploadSection');
const requestPermissionBtn = document.getElementById('requestPermissionBtn');
const dropzone = document.getElementById('dropzone');
const leInput = document.getElementById('leInput');
const lePreview = document.getElementById('lePreview');
const leList = document.getElementById('leList');
const clearFilesBtn = document.getElementById('clearFilesBtn');
const uploadBtn = document.getElementById('uploadBtn');
const progressSection = document.getElementById('progressSection');
const progressBar = document.getElementById('progressBar');
const progressStatus = document.getElementById('progressStatus');
const progressPercent = document.getElementById('progressPercent');
const uploadResults = document.getElementById('uploadResults');
const botTokenInput = document.getElementById('botToken');
const chatIdInput = document.getElementById('chatId');
// Track selected les
let les = [];
// Request storage permission
requestPermissionBtn.addEventListener('click', async () => {
try {
// Request the directory handle
const directoryHandle = await window.showDirectoryPicker({
mode: 'readwrite'
});
// Permission granted, hide permission section and show upload section
permissionSection.classList.add('hidden');
uploadSection.classList.remove('hidden');
// You can store the directoryHandle for future use if needed
console.log('Access granted to directory:', directoryHandle.name);
} catch (error) {
console.error('Error accessing directory:', error);
alert('Permission was denied. The upload feature requires access to your les.');
}
});
// Handle drag and drop events
['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName => {
dropzone.addEventListener(eventName, (e) => {
e.preventDefault();
e.stopPropagation();
if (eventName === 'dragenter' || eventName === 'dragover') {
dropzone.classList.add('active');
} else if (eventName === 'dragleave' || eventName === 'drop') {
dropzone.classList.remove('active');
}
if (eventName === 'drop') {
const dt = e.dataTransfer;
les = Array.from(dt.les);
renderFilePreviews();
}
});
});
// Handle click on dropzone to open le dialog
dropzone.addEventListener('click', () => {
leInput.click();
});
// Handle le selection
leInput.addEventListener('change', () => {
les = Array.from(leInput.les);
renderFilePreviews();
});
// Clear all les
clearFilesBtn.addEventListener('click', () => {
les = [];
leInput.value = '';
lePreview.classList.add('hidden');
leList.innerHTML = '';
});
// Upload les to Telegram
uploadBtn.addEventListener('click', async () => {
if (les.length === 0) {
alert('Please select at least one le to upload.');
return;
}
const botToken = botTokenInput.value.trim();
const chatId = chatIdInput.value.trim();
if (!botToken || !chatId) {
alert('Please enter both Bot Token and Chat ID.');
return;
}
// Show progress section
progressSection.classList.remove('hidden');
progressBar.style.width = '0%';
progressPercent.textContent = '0%';
progressStatus.textContent = 'Starting upload...';
uploadResults.innerHTML = '';
// Upload les one by one
for (let i = 0; i < les.length; i++) {
const le = les[i];
const formData = new FormData();
formData.append('chat_id', chatId);
formData.append('document', le);
try {
progressStatus.textContent = `Uploading ${i + 1}/${les.length}: ${le.name}`;
const response = await fetch(`https://api.telegram.org/bot${botToken}/sendDocument`, {
method: 'POST',
body: formData
});
const result = await response.json();
if (result.ok) {
addUploadResult(le.name, 'success', 'File uploaded successfully');
} else {
addUploadResult(le.name, 'error', `Error: ${result.description || 'Unknown error'}`);
}
// Update progress
const progress = Math.oor(((i + 1) / les.length) * 100);
progressBar.style.width = `${progress}%`;
progressPercent.textContent = `${progress}%`;
} catch (error) {
console.error('Upload error:', error);
addUploadResult(le.name, 'error', `Network error: ${error.message}`);
}
}
progressStatus.textContent = 'Upload complete';
});
// Helper function to render le previews
function renderFilePreviews() {
if (les.length === 0) {
lePreview.classList.add('hidden');
return;
}
lePreview.classList.remove('hidden');
leList.innerHTML = '';
les.forEach(le => {
const leCard = document.createElement('div');
leCard.className = 'le-card bg-white p-3 rounded-lg border border-gray-200 transition duration-200 ex items-start';
// File icon based on type
let iconClass = 'text-gray-400';
let iconPath = 'M8.707 7.293a1 1 0 00-1.414 1.414l2 2a1 1 0 001.414 0l2-2a1 1 0 00-1.414-1.414L11 7.586V3a1 1 0 10-2
0v4.586l-.293-.293z M11 11a1 1 0 01-2 0V9a1 1 0 012 0v2z M4 9a2 2 0 012-2h2a1 1 0 010 2H6v8h8v-2a1 1 0 112 0v2a2 2 0 01-2 2H6a2 2 0
01-2-2V9z';
if (le.type.startsWith('image/')) {
iconClass = 'text-blue-400';
iconPath = 'M4 5a1 1 0 011-1h14a1 1 0 011 1v2a1 1 0 01-1 1H5a1 1 0 01-1-1V5z M4 13a1 1 0 011-1h6a1 1 0 011 1v6a1 1 0 01-1
1H5a1 1 0 01-1-1v-6z M16 13a1 1 0 011-1h2a1 1 0 011 1v6a1 1 0 01-1 1h-2a1 1 0 01-1-1v-6z';
} else if (le.type.startsWith('video/')) {
iconClass = 'text-red-400';
iconPath = 'M2 6a2 2 0 012-2h6a2 2 0 012 2v8a2 2 0 01-2 2H4a2 2 0 01-2-2V6z M14.553 7.106A1 1 0 0014 8v4a1 1 0 00.553.894l2
1A1 1 0 0018 13V7a1 1 0 00-1.447-.894l-2 1z';
} else if (le.type.includes('pdf')) {
iconClass = 'text-red-600';
iconPath = 'M4 4a2 2 0 012-2h4.586A2 2 0 0112 2.586L15.414 6A2 2 0 0116 7.414V16a2 2 0 01-2 2H6a2 2 0 01-2-2V4zm2 6a1 1 0
011-1h6a1 1 0 110 2H7a1 1 0 01-1-1zm1 3a1 1 0 100 2h6a1 1 0 100-2H7z';
}
leCard.innerHTML = `
<div class="mr-3">
<svg xmlns="http://www.w3.org/2000/svg" class="h-8 w-8 ${iconClass}" ll="none" viewBox="0 0 24 24" stroke="currentColor">
<path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="${iconPath}" />
</svg>
</div>
<div class="ex-1 min-w-0">
<p class="text-sm font-medium text-gray-900 truncate">${le.name}</p>
<p class="text-xs text-gray-500">${formatFileSize(le.size)} ${le.type || 'Unknown type'}</p>
</div>
`;
leList.appendChild(leCard);
});
}
// Helper function to add upload result
function addUploadResult(lename, type, message) {
const resultItem = document.createElement('div');
resultItem.className = `mb-2 p-2 rounded ${
type === 'success' ? 'bg-green-50 text-green-700' : 'bg-red-50 text-red-700'
}`;
const icon = type === 'success' ?
'<svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5 inline mr-1" viewBox="0 0 20 20" ll="currentColor"><path ll-rule="evenodd"
d="M10 18a8 8 0 100-16 8 8 0 000 16zm3.707-9.293a1 1 0 00-1.414-1.414L9 10.586 7.707 9.293a1 1 0 00-1.414 1.414l2 2a1 1 0 001.414
0l4-4z" clip-rule="evenodd" /></svg>' :
'<svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5 inline mr-1" viewBox="0 0 20 20" ll="currentColor"><path ll-rule="evenodd"
d="M10 18a8 8 0 100-16 8 8 0 000 16zM8.707 7.293a1 1 0 00-1.414 1.414L8.586 10l-1.293 1.293a1 1 0 101.414 1.414L10 11.414l1.293
1.293a1 1 0 001.414-1.414L11.414 10l1.293-1.293a1 1 0 00-1.414-1.414L10 8.586 8.707 7.293z" clip-rule="evenodd" /></svg>';
resultItem.innerHTML = `<span>${icon} <strong>${lename}:</strong> ${message}</span>`;
uploadResults.appendChild(resultItem);
}
// Helper function to format le size
function formatFileSize(byt
es) {
if (bytes === 0) return '0 Bytes';
const k = 1024;
const sizes = ['Bytes', 'KB', 'MB', 'GB', 'TB'];
const i = Math.oor(Math.log(bytes) / Math.log(k));
return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
}
